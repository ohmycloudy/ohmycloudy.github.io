## unlink files

# create a file
my $f = open "foo",:w;
$f.print(time);
$f.close;

unlink "foo";

## copy and move

# create a file
my $f = open "foo",:w;
$f.print(time);
$f.close;

# copy
my $io = IO::Path.new("foo");
$io.copy("foo2");

# clean up
unlink ("foo2");

# move
$io.rename("foo2");

# clean up
unlink ("foo2");

## open 

my $filename = shift(@*ARGS);

my $fh = open $filename, :a;

my $something;
repeat {
  $something = prompt 'enter something: ';
  $fh.say($something);
} while $something;


## CATCH

my $total = 0;
for @files -> $filename {
    $total += lines($filename.IO).grep(
        { $_ !~~ /<&insigline>/ }
).elems;
 CATCH {
        when X::IO {
            note "Couldn't read $filename";
} }
}
say $total;
