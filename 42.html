## for
.say for ^2020;
.say given 'Wish list';


## when

$_ = 42;
when 42 { say 'find it' };

## for when

for 42, 43, "foo", 44, "bar" {
    when Int       { .say }
    when /:i ^Bar/ { .say }
    default        { say "Not an Int or a Bar" }
}

## for given

for Date.new('2019-01-01') .. Date.new('2019-02-04') {
    printf("%04d%02d%02d\n", .year, .month, .day) given .DateTime
}
